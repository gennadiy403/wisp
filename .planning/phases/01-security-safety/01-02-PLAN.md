def _setup_tooltip():
    global _tooltip_panel, _tooltip_label
    style = AppKit.NSWindowStyleMaskBorderless | AppKit.NSWindowStyleMaskNonactivatingPanel
    panel = AppKit.NSPanel.alloc().initWithContentRect_styleMask_backing_defer_(

def _show_tooltip(text):
    """Show tooltip panel with text. Must be called on main queue."""
    if _tooltip_panel is None:
        return
    _tooltip_label.setStringValue_(text)
    _tooltip_label.sizeToFit()
    h = max(24, int(_tooltip_label.frame().size.height) + 16)
    _tooltip_panel.setFrame_display_(AppKit.NSMakeRect(42, 0, 240, h), True)
    _tooltip_panel.orderFrontRegardless()

def _hide_tooltip():
    """Hide tooltip panel. Must be called on main queue."""
    if _tooltip_panel is not None:
        _tooltip_panel.orderOut_(None)

```python
class HUDClickHandler(AppKit.NSObject):
    def handleClick_(self, sender):
        global _retry_count
        if _hud_error_mode != "error_retryable":
            return
        if _retry_buffer is None:
            return
        _retry_count += 1
        if _retry_count > 3:
            # Max retries exhausted per UI-SPEC
            set_hud(True, mode="error_fatal", tooltip=_tooltip("retry_exhausted"))
            return
        set_hud(True, mode="transcribing", tooltip=_tooltip("retry_attempt", n=_retry_count))
        threading.Thread(target=_retry_transcription, daemon=True).start()
```

**7. Create `_retry_transcription()` function:**

```python
def _retry_transcription():
    """Re-transcribe using _retry_buffer. Called from HUDClickHandler in daemon thread."""
    global _retry_buffer, _retry_count
    buf_copy = _retry_buffer  # local ref -- safe from race
    if buf_copy is None:
        return
    text = _encode_and_transcribe(buf_copy)
    if text is None:
        # Still failing -- show retryable error again
        set_hud(True, mode="error_retryable", tooltip=_tooltip("api_timeout"))
        return
    if not text or text in WHISPER_HALLUCINATIONS:
        print("(empty)", flush=True)
        set_hud(False)
        return
    _retry_count = 0
    # Paste the result (same as normal dictation path)
    paste_text(text)
    set_hud(False)
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Application -> OpenAI API | Audio data crosses network boundary to cloud |
| OpenAI API -> Application | API responses (text, errors) cross back |
| User click -> retry mechanism | Mouse event triggers re-transcription |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-04 | Denial of Service | OpenAI API hang | mitigate | timeout=30.0 on client constructor, max_retries=0 to surface errors immediately (SEC-03) |
| T-01-05 | Denial of Service | Retry loop exhaustion | mitigate | Max 3 retries enforced in HUDClickHandler, then transition to error_fatal |
| T-01-06 | Information Disclosure | API error messages in tooltip | accept | Error messages are generic (no API keys or request details shown to user) |
| T-01-07 | Tampering | _retry_buffer race condition | mitigate | Copy audio_chunks to separate _retry_buffer on error; new recording clears audio_chunks but not _retry_buffer (Pitfall 4 from RESEARCH.md) |
| T-01-08 | Denial of Service | HUD click floods retry | accept | Each retry click replaces previous retry thread; transcription is naturally throttled by API latency |
</threat_model>
